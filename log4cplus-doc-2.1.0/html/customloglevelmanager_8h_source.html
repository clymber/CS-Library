64class CustomLogLevelManager {
65protected:
66 log4cplus::thread::Mutex mtx;
67 bool pushed_methods;
68 std::map<LogLevel,tstring> ll2nm;
69 std::map<tstring,LogLevel> nm2ll;
70
71public:
72 CustomLogLevelManager()
73 : pushed_methods (false)
74 { }
75
76 bool add(LogLevel ll, tstring const &nm)
77 {
78 log4cplus::thread::MutexGuard guard (mtx);
79
80 if (! pushed_methods)
81 {
82 pushed_methods = true;
83 getLogLevelManager().pushToStringMethod(customToStringMethod);
84 getLogLevelManager().pushFromStringMethod(customFromStringMethod);
85 }
86
87 auto i = ll2nm.lower_bound(ll);
88 if( ( i != ll2nm.end() ) && ( i->first == ll ) && ( i->second != nm ) )
89 return false;
90
91 auto j = nm2ll.lower_bound(nm);
92 if( ( j != nm2ll.end() ) && ( j->first == nm ) && ( j->second != ll ) )
93 return false;
94
95 // there is no else after return
96 ll2nm.insert( i, std::make_pair(ll, nm) );
97 nm2ll.insert( j, std::make_pair(nm, ll) );
98 return true;
99 }
100
101 bool remove(LogLevel ll, tstring const &nm)
102 {
103 log4cplus::thread::MutexGuard guard (mtx);
104
105 auto i = ll2nm.find(ll);
106 auto j = nm2ll.find(nm);
107 if( ( i != ll2nm.end() ) && ( j != nm2ll.end() ) &&
108 ( i->first == j->second ) && ( i->second == j->first ) ) {
109 ll2nm.erase(i);
110 nm2ll.erase(j);
111
112 return true;
113 }
114
115 // there is no else after return
116 return false;
117 }
118
119protected:
120 tstring const & customToStringMethodWorker(LogLevel ll)
121 {
122 log4cplus::thread::MutexGuard guard (mtx);
123 auto i = ll2nm.find(ll);
124 if( i != ll2nm.end() )
125 return i->second;
126
127 return internal::empty_str;
128 }
129
130 LogLevel customFromStringMethodWorker(const tstring& nm)
131 {
132 log4cplus::thread::MutexGuard guard (mtx);
133 auto i = nm2ll.find(nm);
134 if( i != nm2ll.end() )
135 return i->second;
136
137 return NOT_SET_LOG_LEVEL;
138 }
139
140 LOG4CPLUS_PRIVATE static tstring const & customToStringMethod(LogLevel ll);
141 LOG4CPLUS_PRIVATE static LogLevel customFromStringMethod(const tstring& nm);
142};
