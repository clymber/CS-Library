140struct socket_holder
141{
142 os_socket_type sock;
143
144 socket_holder()
145 : sock(INVALID_OS_SOCKET_VALUE)
146 { }
147
148 socket_holder(os_socket_type s)
149 : sock(s)
150 { }
151
152 ~socket_holder()
153 {
154 socket_closer()(sock);
155 }
156
157 void
158 reset(os_socket_type s = INVALID_OS_SOCKET_VALUE)
159 {
160 if (sock != INVALID_OS_SOCKET_VALUE)
161 socket_closer()(sock);
162
163 sock = s;
164 }
165
166 os_socket_type
167 detach()
168 {
169 os_socket_type s = sock;
170 sock = INVALID_OS_SOCKET_VALUE;
171 return s;
172 }
