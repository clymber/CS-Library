226#define LOG4CPLUS_MACRO_BODY(logger, logEvent, logLevel) \
227 LOG4CPLUS_SUPPRESS_DOWHILE_WARNING() \
228 do { \
229 log4cplus::Logger const & _l \
230 = log4cplus::detail::macros_get_logger (logger); \
231 if LOG4CPLUS_MACRO_LOGLEVEL_PRED ( \
232 _l.isEnabledFor (log4cplus::logLevel), logLevel) { \
233 LOG4CPLUS_MACRO_INSTANTIATE_OSTRINGSTREAM (_log4cplus_buf); \
234 _log4cplus_buf << logEvent; \
235 LOG4CPLUS_MACRO_LOG_LOCATION (_logLocation); \
236 log4cplus::detail::macro_forced_log (_l, \
237 log4cplus::logLevel, _log4cplus_buf.str(), \
238 _logLocation.file_name (), \
239 _logLocation.line (), \
240 _logLocation.function_name ()); \
241 } \
242 } while (false) \
243 LOG4CPLUS_RESTORE_DOWHILE_WARNING()
